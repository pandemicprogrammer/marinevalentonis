Python Weather API App

REST APIs, JSON, environment variables, CSV


import requests
import csv
from keys import API_KEY


CITY = 'Shanghai'
COUNTRY = 'CN'
UNITS = 'imperial'

res = requests.get(f'http://api.openweathermap.org/data/2.5/weather?q={CITY},{COUNTRY},&units={UNITS}&appid={API_KEY}')
data = res.json()

print(data)

weather = data['weather'][0]['main']
temp = data['main']['temp']
wind_speed = data['wind']['speed']

header = ['city', 'country_code', 'weather', 'wind_speed', 'temp'] #CSV header columns
info = [CITY, COUNTRY, wind_speed, weather, temp]

with open('weather.csv', 'w') as f:
    #create csv
    writer = csv.writer(f)
    #writes header row
    writer.writerow(header)
    #writes info row
    writer.writerow(info)

                                this.calculateMetalWeight();

                                this.metalWeight =
                                  (this.waxWeight + this.waxWeight * this.buttonSize) *

                                  this.pennyWeight = this.metalWeight * 0.643015
                                  },

                                const target = document.getElementById("result");
                                if (target.style.display !== "block") {
                                  target.style.display = "block";
                                  } 
                                }
                              
                            
                            },

                            data() {
